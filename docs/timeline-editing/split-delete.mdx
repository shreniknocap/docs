<div id="ios-tab">
    <Tabs>
        <Tab title="iOS">
            **Split a clip**

            1. Tap on the timeline
            2. Scroll to the desired location
            3. Tap **Split**

            **Delete a clip**

            1. Tap on the timeline
            2. Tap **Split** at the beginning and end of your desired location
            3. Tap on the segment to be removed and tap **Delete**
        </Tab>

        <Tab title="Desktop">
            **Split a clip**

            1. Click on the timeline
            2. Click **Split** at the beginning and end of your desired location
            3. Click on the segment to be removed and click **Delete**

            **Delete a clip**

            1. Click on the timeline
            2. Click **Split** at the beginning and end of your desired location
            3. Click on the segment to be removed and click **Delete**
        </Tab>
